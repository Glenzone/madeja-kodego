Height Converter

Converts height in foot-inches unit to meters.

ft in m

Solution


function convert() {
  document.getElementById('validationMsg').innerHTML = "";
  let inputFeet = parseFloat(document.getElementById('inputFeet').value);
  let inputInches = parseFloat(document.getElementById('inputInches').value);     

  let metersFromFeet = inputFeet * 0.3048;
  let metersFromInches = inputInches * 0.0254;
  let inputMeters = Math.round((metersFromFeet + metersFromInches) * 100) / 100;
  document.getElementById('inputMeters').value = inputMeters;

  //validation
  if(isNaN(inputMeters)) {
    document.getElementById('validationMsg').innerHTML = `Uhm.. The app does not 
    support conversion of other characters to numbers.</p> <p> Please 
    input a number.`;
    document.getElementById('inputMeters').value = 0;
  } else if (inputFeet % 1 !== 0) {
    document.getElementById('validationMsg').innerHTML = `Oops.. Check your 
    <strong>ft</strong> value.</p> <p>Shouldn't decimal 
    values be converted to inches?`;
    document.getElementById('inputMeters').value = 0;
  };
}
