Greeting Generator

Generates greeting based on the inputted day in a week.

Input a day

Kamusta Ka?

Solution


function greet() {
  let inputDay = document.getElementById('inputDay').value;
  let lowerDay = inputDay.toLowerCase();

  switch(lowerDay) {
    case "monday":
      document.getElementById('dayGreeting').innerHTML = `Nǐn Hǎo`;
      break;
    case "tuesday":
      document.getElementById('dayGreeting').innerHTML = `Guten Tag`;
      break;
    case "wednesday":
      document.getElementById('dayGreeting').innerHTML = `Bonjour!`;
      break;
    case "thursday":
      document.getElementById('dayGreeting').innerHTML = `Anyoung Haseyo`;
      break;
    case "friday":
      document.getElementById('dayGreeting').innerHTML = `Olá`;
      break;
    case "saturday":
      document.getElementById('dayGreeting').innerHTML = `Asalaam Alaikum`;
      break;
    case "sunday":
      document.getElementById('dayGreeting').innerHTML = `Konnichiwa`;
      break;
    default:
      document.getElementById('dayGreeting').innerHTML = `Kamusta Ka?`;
      break;
  }
};
